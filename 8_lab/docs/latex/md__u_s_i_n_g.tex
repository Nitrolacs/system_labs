Для запуска сервера использовать команду\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{./server\ [-\/l\ log\_file]\ [-\/t\ timeout]\ [-\/f\ filesystem\_check]}

\end{DoxyCode}

\begin{DoxyCode}{0}
\DoxyCodeLine{./client\ -\/a\ a\ -\/b\ b\ -\/c\ c\ [-\/d\ d]\ [-\/l\ log\_file]\ [-\/t\ timeout]\ [-\/f\ filesystem\_check]}

\end{DoxyCode}
